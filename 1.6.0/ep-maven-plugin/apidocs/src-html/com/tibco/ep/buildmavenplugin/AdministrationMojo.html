030package com.tibco.ep.buildmavenplugin;
031
032import java.util.Map;
033
034import org.apache.maven.plugin.MojoExecutionException;
035import org.apache.maven.plugins.annotations.Mojo;
036import org.apache.maven.plugins.annotations.Parameter;
037
038/**
039 * <p>Administer nodes</p>
040 * <p>Example use in pom.xml:</p>
041 * <img src="uml/administer-nodes.svg" alt="pom">
042 * 
043 */
044@Mojo(name = "administer-nodes", threadSafe = true)
045public class AdministrationMojo extends BaseExecuteMojo {
046
047    /**
048     * <p>Administration command to execute</p>
049     *
050     * <p>Example use in pom.xml:</p>
051     * <img src="uml/administer-nodes-command.svg" alt="pom">
052     * 
053     * @since 1.0.0
054     */
055    @Parameter(required=true)
056    String command;
057
058    /**
059     * <p>Administration target to execute</p>
060     * 
061     * <p>Example use in pom.xml:</p>
062     * <img src="uml/administer-nodes-target.svg" alt="pom">
063     * 
064     * @since 1.0.0
065     */
066    @Parameter(required=true)
067    String target;
068
069    /**
070     * <p>Administration port - overrides servicename</p>
071     * 
072     * <p>Example use in pom.xml:</p>
073     * <img src="uml/administer-nodes-adminport.svg" alt="pom">
074     * 
075     * @since 1.0.0
076     */
077    @Parameter
078    Integer adminport;
079
080    /**
081     * <p>Hostname - default is localhost.  Only applies when adminport is set</p>
082     * 
083     * <p>Example use in pom.xml:</p>
084     * <img src="uml/administer-nodes-hostname.svg" alt="pom">
085     * 
086     * @since 1.0.0
087     */
088    @Parameter
089    String hostname;
090    
091    /**
092     * <p>Administration command arguments</p>
093     * 
094     * <p>Example use in pom.xml:</p>
095     * <img src="uml/administer-nodes-arguments.svg" alt="pom">
096     * 
097     * @since 1.0.0
098     */
099    @Parameter
100    Map<String, String> arguments;
101    
102    /**
103     * <p>Servicename to select which nodes to run the administration command on.</p>
104     * 
105     * <p>If not set, test cases will be run on all the nodes in the cluster.</p>
106     * 
107     * <p>Example use in pom.xml:</p>
108     * <img src="uml/administer-nodes-serviceName.svg" alt="pom">
109     * 
110     * @since 1.0.0
111     */
112    @Parameter(defaultValue="${project.artifactId}")
113    String serviceName;
114    
115    /**
116     * <p>Skip executing the administration command</p>
117     * 
118     * <p>Example use in pom.xml:</p>
119     * <img src="uml/administer-nodes-skip.svg" alt="pom">
120     * 
121     * @since 1.0.0
122     */
123    @Parameter(defaultValue="false")
124    boolean skip;
125    
126    public void execute() throws MojoExecutionException {
127        getLog().debug("Administer nodes");
128
129        prechecks();
130
131        if (skip) {
132            getLog().info("Administrator command is skipped");
133            return;
134        }
135        
136        initializeAdministration(true);
137        if (adminport == null) {
138            runAdministrationCommand(serviceName, userName, password, command, target, arguments, true);
139        } else {
140            runAdministrationCommand(adminport, hostname, userName, password, command, target, arguments, true);
141        }
142    }
143
144}
